                              type="text" 
                              value={lobData[lob].calls.sla} 
                              onChange={(e) => updateLobMetric(lob, 'calls', 'sla', e.target.value)}
                              className={`w-full mt-1 px-2 py-1 text-xs font-semibold border rounded outline-none focus:border-indigo-500 focus:ring-1 focus:ring-indigo-500 ${t.inputBg} ${t.inputBorder} ${t.textMain}`} 
                            />
                          </div>
                        </div>
                      </div>

                    </div>
                  ))}
                </div>

                <div className="space-y-3">
                  <h4 className={`text-[11px] font-bold uppercase tracking-wide border-b pb-1 ${isDark ? 'text-neutral-400 border-neutral-800' : 'text-slate-500 border-slate-200'}`}>Email Productivity</h4>
                  <div className="grid grid-cols-2 gap-3">
                    {['closed', 'assigned', 'replied', 'sent'].map(field => (
                      <div key={field}>
                        <label className={`text-[10px] font-medium uppercase ${isDark ? 'text-neutral-500' : 'text-slate-500'}`}>{field}</label>
                        <input 
                          type="number" 
                          value={emailData[field]} 
                          onChange={(e) => setEmailData(prev => ({...prev, [field]: parseInt(e.target.value) || 0}))}
                          className={`w-full mt-1 px-2.5 py-1.5 text-xs font-semibold border rounded outline-none focus:border-indigo-500 focus:ring-1 focus:ring-indigo-500 ${t.inputBg} ${t.inputBorder} ${t.textMain}`} 
                        />
                      </div>
                    ))}
                  </div>
                </div>

                <div className="space-y-6 pb-8">
                  <h4 className={`text-[11px] font-bold uppercase tracking-wide border-b pb-1 ${isDark ? 'text-neutral-400 border-neutral-800' : 'text-slate-500 border-slate-200'}`}>Shift Operations Log</h4>
                  
                  {['mitigations', 'causes', 'keynotes'].map(field => {
                    const data = opsLog[field];
                    const presets = opsLogPresets[field];
                    
                    const togglePreset = (preset) => {
                      setOpsLog(prev => ({
                        ...prev,
                        [field]: {
                          ...prev[field],
                          selected: prev[field].selected.includes(preset)
                            ? prev[field].selected.filter(p => p !== preset)
                            : [...prev[field].selected, preset]
                        }
                      }));
                    };

                    const dotColor = field === 'mitigations' ? 'bg-amber-400' : field === 'causes' ? 'bg-red-500' : 'bg-emerald-500';
                    const title = field === 'mitigations' ? 'Mitigations Made' : field === 'causes' ? 'Causes of Abandonment' : 'Keynotes';

                    return (
                      <div key={field} className="space-y-2">
                        <label className={`text-[10px] font-medium uppercase flex items-center gap-1.5 ${isDark ? 'text-neutral-400' : 'text-slate-500'}`}>
                          <div className={`w-1.5 h-1.5 rounded-full ${dotColor}`}></div> {title}
                        </label>
                        
                        {/* Multi-select Pills */}
                        <div className="flex flex-wrap gap-1.5">
                          {presets.map(preset => {
                            const isSelected = data.selected.includes(preset);
                            return (
                              <button
                                key={preset}
                                onClick={() => togglePreset(preset)}
                                className={`px-2 py-1 text-[10px] font-semibold rounded border transition-colors ${
                                  isSelected 
                                    ? (isDark ? 'bg-indigo-500/20 text-indigo-300 border-indigo-500/50' : 'bg-indigo-50 text-indigo-700 border-indigo-200')
                                    : (isDark ? 'bg-transparent text-neutral-500 border-neutral-700 hover:border-neutral-500' : 'bg-transparent text-slate-500 border-slate-300 hover:border-slate-400')
                                }`}
                              >
                                {preset}
                              </button>
                            );
                          })}
                          
                          {/* Custom Edit Toggle */}
                          <button
                            onClick={() => setOpsLog(prev => ({...prev, [field]: {...prev[field], showCustom: !prev[field].showCustom}}))}
                            className={`px-2 py-1 text-[10px] font-semibold rounded border border-dashed transition-colors ${
                              data.showCustom
                                ? (isDark ? 'bg-neutral-800 text-white border-neutral-600' : 'bg-slate-200 text-slate-800 border-slate-400')
                                : (isDark ? 'bg-transparent text-neutral-500 border-neutral-700 hover:border-neutral-500' : 'bg-transparent text-slate-500 border-slate-300 hover:border-slate-400')
                            }`}
                          >
                            + Custom Edit
                          </button>
                        </div>

                        {/* Custom Textarea */}
                        {data.showCustom && (
                          <textarea
                            value={data.custom}
                            onChange={(e) => setOpsLog(prev => ({...prev, [field]: {...prev[field], custom: e.target.value}}))}
                            placeholder={`Type custom ${field} note here...`}
                            className={`w-full mt-2 px-2.5 py-2 text-xs font-medium border rounded outline-none focus:border-indigo-500 focus:ring-1 focus:ring-indigo-500 min-h-[50px] ${t.inputBg} ${t.inputBorder} ${t.textMain}`}
                          />
                        )}
                      </div>
                    );
                  })}
                </div>

              </div>
            </div>
          </div>
        )}

        {/* Toast Notification System */}
        {toast.show && (
          <div className={`fixed bottom-6 right-6 px-4 py-3 rounded-lg shadow-lg border flex items-center gap-2 z-[200] animate-in slide-in-from-bottom-5 ${
            toast.type === 'success' ? (isDark ? 'bg-neutral-900 border-green-900/50 text-green-400' : 'bg-white border-emerald-200 text-emerald-800') : toast.type === 'error' ? (isDark ? 'bg-neutral-900 border-red-900/50 text-red-400' : 'bg-white border-red-200 text-red-800') : (isDark ? 'bg-neutral-900 border-neutral-700 text-neutral-200' : 'bg-white border-slate-200 text-slate-800')
          }`}>
            {toast.type === 'success' ? <CheckCircle2 className={`w-4 h-4 ${isDark ? 'text-green-500' : 'text-emerald-500'}`} /> : <Activity className={`w-4 h-4 ${isDark ? 'text-neutral-500' : 'text-slate-500'}`} />}
            <span className="text-xs font-semibold">{toast.message}</span>
          </div>
        )}

      </div>
    </div>
  );
}
